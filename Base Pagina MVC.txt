CREATE TABLE productos (
    idArt INT PRIMARY KEY IDENTITY(1,1),
    nombre NVARCHAR(80),
    descripcion NVARCHAR(100),
    precio DECIMAL,
    imagen NVARCHAR(255));
GO

CREATE TABLE usuarios (
    id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    nomCompleto NVARCHAR(40),
    correo NVARCHAR(35),
    contra NVARCHAR(25),
    rol NVARCHAR(50)
);
Go

CREATE TABLE pedidos (
    idPedido INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    idArt INT NOT NULL,
    nombre NVARCHAR(40),
    cantidad INT,
    precio DECIMAL,
    totalGeneral DECIMAL,
    fechaCompra DATE,
    idUsuario int
);
Go
	
CREATE TABLE carrito (
    idArt INT,
    nombre VARCHAR(40),
    cantidad INT,
    precio DECIMAL,
    total DECIMAL,
    PRIMARY KEY (idArt)
);
Go
